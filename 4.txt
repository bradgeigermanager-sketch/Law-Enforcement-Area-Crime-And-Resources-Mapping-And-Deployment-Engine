);
-- NEW: Rule engine matrix matching incident types to responder profilesCREATE TABLE notification_routing_rules (
    rule_id INT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
    trigger_category VARCHAR(50) NOT NULL, -- e.g., 'Violent', 'Cyber', 'Statutory/Regulatory'
    min_escalation_tier VARCHAR(20) NOT NULL, -- e.g., 'TIER_2_REGIONAL', 'TIER_3_CRITICAL'
    target_role_id INT REFERENCES responder_roles(role_id),
    auto_escalate_on_sensor_failure BOOLEAN DEFAULT TRUE,
    is_active BOOLEAN DEFAULT TRUE
);
CREATE INDEX idx_routing_rules ON notification_routing_rules(trigger_category, min_escalation_tier) WHERE is_active = TRUE;

for idx, row in forecast_df.iterrows():

    vol = row['Predicted_Volume']
    cat = row['Category']
    tier = "TIER_3_CRITICAL" if vol >= 7.0 else ("TIER_2_REGIONAL" if vol >= 3.5 else "TIER_1_LOCAL")
    escalation_tiers.append(tier)
    
    # 2. Dynamic Routing Rule Set Verification Block
    if tier == "TIER_3_CRITICAL" and cat == "Violent":
        assigned = "TACTICAL"
    elif cat == "Cyber":
        assigned = "CYBER"
    elif tier == "TIER_3_CRITICAL" or tier == "TIER_2_REGIONAL" and cat == "Property":
        assigned = "TACTICAL"
    elif cat == "Statutory/Regulatory" and tier != "TIER_3_CRITICAL":
        assigned = "MUNICIPAL"
    else:
        assigned = "INFRA_ENG" # Default infrastructure maintenance routing assignment

    matched_meta = roles_df[roles_df['Code'] == assigned].iloc
    routing_roles.append(matched_meta['Role'])
    dispatch_endpoints.append(matched_meta['Endpoint'])
