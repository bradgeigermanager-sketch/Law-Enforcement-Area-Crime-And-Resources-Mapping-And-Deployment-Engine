Here is the updated, unified UI and Numeracy Engine written in Python using streamlit.
This version integrates two new analytical panels: a Statistical Estimation Module that utilizes a Poisson distribution to calculate the mathematical probability of a specific number of offenses occurring within a custom time window, and a Historical Trend Comparison Module that tracks trajectory shifts year-over-year.
## Updated Python UI Engine (app.py)
Run pip install streamlit pandas plotly scipy numpy to install the requirements before launching the engine via streamlit run app.py.

import streamlit as stimport pandas as pdimport numpy as npimport plotly.express as pximport plotly.graph_objects as gofrom scipy import stats
# 1. Mock Data Generator tracking multi-year legal trends
@st.cache_datadef load_mock_legal_data():
    np.random.seed(42)
    categories = ['Property', 'Violent', 'Cyber', 'Statutory/Regulatory']
    statutes = {
        'Property': ['Larceny-Theft', 'Burglary', 'Motor Vehicle Theft', 'Fraud'],
        'Violent': ['Aggravated Assault', 'Robbery', 'Homicide'],
        'Cyber': ['Identity Theft', 'Phishing Networks', 'Ransomware Deployment'],
        'Statutory/Regulatory': ['Traffic Infractions', 'Tax Evasion', 'Licensing Violations']
    }
    
    data = []
    # Generates 5 years of historical baseline trends up to 2026
    for year in range(2022, 2027):
        for cat in categories:
            for stat in statutes[cat]:
                if cat == 'Property':
                    base = np.random.randint(8000, 15000)
                elif cat == 'Statutory/Regulatory':
                    base = np.random.randint(12000, 20000)
                elif cat == 'Cyber':
                    base = np.random.randint(3000, 7000) + (year - 2022) * 1500 # Simulating structural upward shift
                else: 
                    base = np.random.randint(1500, 4500)
                
                data.append({
                    "Year": year,
                    "Statute": stat,
                    "Category": cat,
                    "Charges Filed": int(base * np.random.uniform(0.85, 1.15))
                })
    return pd.DataFrame(data)

# ==========================================# TAB 2: HISTORICAL TREND COMPARISONS# ==========================================with tab2:

# ==========================================# TAB 3: STATISTICAL ESTIMATION (PREDICTIVE)# ==========================================with tab3:
    st.header("Poisson Event Probability & Estimation Module")
    st.markdown("Because criminal offenses arrive independently and at a known average rate, we can apply a **Poisson Distribution** model to calculate arrival probabilities over any given timeframe.")
    
    # 1. Parameter Settings for Estimation Block
    col_param1, col_param2, col_param3 = st.columns(3)
    with col_param1:
        target_statute = st.selectbox("Select Target Statute for Predictive Modeling", sorted(year_df['Statute'].unique()))
    with col_param2:
        time_window = st.selectbox("Define Operational Time Window", ["Daily", "Weekly", "Monthly"])
    with col_param3:
        n_events = st.number_input("Target Volume Threshold to Evaluate (k events)", min_value=0, value=25)
        
    # Extract annual count to build out arrival rate lambda
    annual_count = year_df[year_df['Statute'] == target_statute]['Charges Filed'].values[0]
    
    # Time-window denominator adjustment
    time_denominators = {"Daily": 365, "Weekly": 52, "Monthly": 12}
    denom = time_denominators[time_window]
    
    # Lambda = average rate of occurrences per selected time window
    lam = annual_count / denom
    
    # 2. Probability Computation Engines
    prob_exact = stats.poisson.pmf(n_events, lam) * 100
    prob_less_equal = stats.poisson.cdf(n_events, lam) * 100
    prob_greater = (1 - stats.poisson.cdf(n_events, lam)) * 100
    
    # Results Visual Display Block
    st.subheader(f"Predictive Insights Matrix: {target_statute}")
    st.info(f"**Baseline Parameter:** Historically, this system records an average of **{lam:.2f}** charges of *{target_statute}* per **{time_window.lower()}** window.")
    
    est_kpi1, est_kpi2, est_kpi3 = st.columns(3)
    est_kpi1.metric(f"Probability of EXACTLY {n_events} Crimes", f"{prob_exact:.3f}%")
    est_kpi2.metric(f"Probability of ≤ {n_events} Crimes", f"{prob_less_equal:.2f}%")
    est_kpi3.metric(f"Probability of MORE THAN {n_events} Crimes", f"{prob_greater:.2f}%")
    
    # 3. Probability Mass Function Distribution Graphing
    # Generate an evaluation array spanning from 0 to twice the expected lambda value
    x_axis = np.arange(0, max(int(lam * 2.5), n_events + 10))
    pmf_values = stats.poisson.pmf(x_axis, lam)
    
    dist_df = pd.DataFrame({'Events (k)': x_axis, 'Probability': pmf_values})
    
    fig_dist = px.bar(dist_df, x='Events (k)', y='Probability', title=f"Poisson Probability Distribution Curve for {target_statute} ({time_window})",
                      labels={'Probability': 'Likelihood of Occurrence (Proportion)'})
    
    # Add an indicator line showing the user's evaluated target volume threshold
    fig_dist.add_vline(x=n_events, line_width=3, line_dash="dash", line_color="orange", 
                       annotation_text=f" Target Threshold: {n_events}")
    
    st.plotly_chart(fig_dist, use_container_width=True)

## Architectural Breakdown of New Modules

* Longitudinal Data Matching (Tab 2): The visualization system handles multidimensional scaling by computing historical parameters over multiple sequential years. It automatically maps lines using plotly.express.line while concurrently computing and indexing Year-over-Year growth/decay velocities via pandas .pct_change().
* Poisson Probability Engine (Tab 3): Criminal activity acts mathematically as an arrival process. If crimes happen independently at a known constant rate ($\lambda$), the system calculates the explicit probability of seeing $k$ occurrences in a time unit:
$$\text{P}(X = k) = \frac{\lambda^k e^{-\lambda}}{k!}$$ 
The UI applies scipy.stats.poisson to compute Probability Mass Functions (pmf) and Cumulative Distribution Functions (cdf) to generate exact risk vectors.
